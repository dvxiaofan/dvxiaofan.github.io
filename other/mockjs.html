    # 利用Mock.js本地批量生成模拟JSON数据

    # 导入Mock.js

    在 HTML 页面里引入

    <script src="http://cdn.staticfile.org/Mock.js/1.0.0/mock-min.js"></script>

    # 批量生成模拟数据

    let Random = Mock.Random;
Random.date();
let dataMock = Mock.mock({
    'dataList|50': [{
        'id|1000001-200000': 666666,
        'name': '@cname',
        'email': '@email',
        'date': '@date',
        'str': '@csentence'
    }]
});

console.log(dataMock);

    # 查看结果数据
